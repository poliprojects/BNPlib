\begin{frame}[fragile]{Algorithms}
Commmon structure:
\begin{small}
\begin{verbatim}
void step(){
    sample_allocations();
    sample_unique_values();
}

void run(BaseCollector* collector){      
    print_startup_message();
    initialize();
    unsigned int iter = 0;
    while(iter < maxiter){
        step();    
        if(iter >= burnin){
            save_state(collector, iter);
        }
    iter++;
    }
    print_ending_message();
}
\end{verbatim}
\end{small}
\end{frame}
